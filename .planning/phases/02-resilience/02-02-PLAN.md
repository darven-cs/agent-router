    ```go
    import "sync"

    // SharedUpstreams is thread-safe shared state for upstreams that both
    // TUI and ProxyHandler reference. Protected by RWMutex.
    type SharedUpstreams struct {
        mu        sync.RWMutex
        upstreams []*Upstream
    }

    func NewSharedUpstreams(upstreams []*Upstream) *SharedUpstreams {
        return &SharedUpstreams{upstreams: upstreams}
    }

    func (s *SharedUpstreams) GetAll() []*Upstream {
        s.mu.RLock()
        defer s.mu.RUnlock()
        result := make([]*Upstream, len(s.upstreams))
        copy(result, s.upstreams)
        return result
    }

    func (s *SharedUpstreams) Add(u *Upstream) {
        s.mu.Lock()
        defer s.mu.Unlock()
        s.upstreams = append(s.upstreams, u)
    }

    func (s *SharedUpstreams) Update(name string, u *Upstream) bool {
        s.mu.Lock()
        defer s.mu.Unlock()
        for i, us := range s.upstreams {
            if us.Name == name {
                s.upstreams[i] = u
                return true
            }
        }
        return false
    }

    func (s *SharedUpstreams) Delete(name string) bool {
        s.mu.Lock()
        defer s.mu.Unlock()
        for i, us := range s.upstreams {
            if us.Name == name {
                s.upstreams = append(s.upstreams[:i], s.upstreams[i+1:]...)
                return true
            }
        }
        return false
    }
    ```

    // Global shared upstreams state (D-25)
    var sharedUpstreams *upstream.SharedUpstreams

    func main() {
        // ... existing setup ...

        // Initialize shared upstreams from config
        sharedUpstreams = upstream.NewSharedUpstreams(cfg.Upstreams)

        // Create TUI with shared state getter
        model := tui.NewModel(serviceName, version, port, sharedUpstreams.GetAll())

        // Run TUI and handle messages
        p := tea.NewGoroutine(model)
        if _, err := p.Run(); err != nil {
            fmt.Fprintf(os.Stderr, "TUI error: %v\n", err)
        }

        // Handle messages from TUI
        for {
            select {
            case msg := <-p.Messages():
                switch m := msg.(type) {
                case tui.UpstreamAdded:
                    sharedUpstreams.Add(m.Upstream)
                    // Also update LoadBalancer
                    lb.AddUpstream(m.Upstream)
                case tui.UpstreamUpdated:
                    if m.OldName != "" && m.OldName != m.Upstream.Name {
                        sharedUpstreams.Delete(m.OldName)
                        lb.DeleteUpstream(m.OldName)
                    }
                    sharedUpstreams.Update(m.Upstream.Name, m.Upstream)
                    lb.UpdateUpstream(m.Upstream)
                case tui.UpstreamDeleted:
                    sharedUpstreams.Delete(m.Name)
                    lb.DeleteUpstream(m.Name)
                }
            default:
                // No more messages
                goto afterMessages
            }
        }
    afterMessages:

        // Graceful shutdown with 10s timeout (D-21)
        fmt.Println("\nShutting down gracefully...")
        ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
        defer cancel()

        if err := server.Shutdown(ctx); err != nil {
            fmt.Fprintf(os.Stderr, "Shutdown error (forcing close): %v\n", err)
            server.Close()
        }
        fmt.Println("Goodbye!")
    }
