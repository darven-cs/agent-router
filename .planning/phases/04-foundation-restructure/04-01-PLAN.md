       ```go
       type LoadBalancer struct {
           upstreams []*Upstream
           primary   *Upstream    // nil = auto hash mode
           mu        sync.RWMutex // protects primary field (NOT the upstreams field -- upstreams is only written by doReload which replaces the whole lb value)
       }

       func (lb *LoadBalancer) SetPrimary(u *Upstream) {
           lb.mu.Lock()
           defer lb.mu.Unlock()
           lb.primary = u
       }

       func (lb *LoadBalancer) ClearPrimary() {
           lb.mu.Lock()
           defer lb.mu.Unlock()
           lb.primary = nil
       }

       func (lb *LoadBalancer) GetPrimary() *Upstream {
           lb.mu.RLock()
           defer lb.mu.RUnlock()
           return lb.primary
       }

       func (lb *LoadBalancer) SelectForRequest(hashInput string) *Upstream {
           lb.mu.RLock()
           primary := lb.primary
           lb.mu.RUnlock()
           if primary != nil {
               for _, u := range lb.upstreams {
                   if u == primary && u.Enabled {
                       return primary
                   }
               }
           }
           return lb.Select(hashInput)
       }
       ```

       ```go
       func (s *SharedUpstreams) ReplaceAll(upstreams []*Upstream) {
           s.mu.Lock()
           defer s.mu.Unlock()
           s.upstreams = upstreams
       }
       ```
